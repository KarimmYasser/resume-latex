% Loop through all experiences
\newcommand{\experienceItem}[4]{%
    \begin{samepage}%
    \begin{twocolentry}{#3}%
    \textbf{\large #1} - #2%
    \end{twocolentry}%
    \vspace{0.1 cm}%
    \begin{onecolentry}%
    \begin{highlights}%
    #4%
    \end{highlights}%
    \end{onecolentry}%
    \end{samepage}%
    \vspace{0.2cm}%
}

% Process experience list with delimiter
\newcommand{\processExperiences}[1]{%
  \processExperiencesAux#1\relax
}

\long\def\processExperiencesAux#1#2#3#4|||#5\relax{%
  \ifx\relax#1\relax
    % Stop if nothing left
  \else
    \experienceItem{#1}{#2}{#3}{#4}%
    \ifx\relax#5\relax
      % No more entries — stop
    \else
      \expandafter\processExperiencesAux#5\relax
    \fi
  \fi
}

\expandafter\processExperiences\expandafter{\experienceList}
